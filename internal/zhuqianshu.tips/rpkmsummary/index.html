一个使用multiBamSummary计算区域rpkm的脚本

整理我使用的一些命令成一个bash脚本，还没有测试：
脚本位置：/home/share/codes/qszhu/rpkmSummary.sh

#!/bin/bash
set -e
help()
{ cat << HELP
    This is a shell script calculate the 
	本程序使用Bam文件计算特定区域（bed文件）内的rpkm，以及与input的比较
	使用范例：
		rpkmSummary -b genes.bed -t treatment.bam -c input.bam -o $odir/$key ...
	必须参数：
		-b 输入的需要计算的区域，bed文件 【必须提前建立index】
		-t 输入的ChIP-seq比对，Bam文件
		-o 输出文件的前缀，可带输出路径
		
	可选参数：
		-c 输入的input或IgG比对的，bam文件
		-g 输入的参考基因组染色体长度，chromSizes文件，或genome ID [默认：mm10,即/home/share/chrom_sizes/mm10.chrom.sizes]
		--id 提供bed文件的区域的ID列，将在输出文件中第5列记录区域ID [应大于3，默认0，不输出ID]
		--tmp 中间结果temp文件存放目录 [默认：/tmp]
		-p 使用线程数 [默认：8]
		-v verbose模式，保留temp文件 [默认删除temp]
		
	-c input.bam使用的可选参数：
		--operation {ratio,logXratio,reciprocal_ratio,subtract,logXsubtract} [default: log2ratio]
			ratio: (treatment rpkm + pseudocount)/(input rpkm + pseudocount)
			logXratio: log((treatment rpkm + pseudocount)/(input rpkm + pseudocount))
			reciprocal_ratio: as ratio, the reciprocal ratio returns the the negative of the inverse of the ratio if the ratio is less than 0
			subtract: (treatment rpkm) - (input rpkm)
			logXsubtract: log((treatment rpkm) - (input rpkm) + pseudocount), if treatment less then input, result is log(pseudocount)
		--pseudocount PSEUDOCOUNT [default: log2ratio]
			A small number to avoid x/0 or log(0)
		--logx {2,10...} [default: 2]
			可自定义log ratio或log subtract的对数底数，默认为log2
			若--operation的参数为log2ratio 或 log10ratio等，该参数无用
		
	multiBamSummary的已默认设置参数：
		--ignoreDuplicates --samFlagInclude 2 --samFlagExclude 1024 --maxFragmentLength 1000 --centerReads
		可使用任意参数覆盖
HELP
exit 0
}

p=8
multiBamSummary="--ignoreDuplicates --samFlagInclude 2 --samFlagExclude 1024 --maxFragmentLength 1000 --centerReads"
declare -A multiBamSummary_operations
#multiBamSummary_operations=(["--ignoreDuplicates"]="" ["--samFlagInclude"]="2" ["--samFlagExclude"]="1024" ["--maxFragmentLength"]="1000" ["--centerReads"]="")
chromsize=mm10
operation=logXratio
pseudocount=1
logx=2
tmpdir=/tmp
hasid=0
while [ -n "$1" ]; do
case "$1" in
    -t) in_file=$2; shift 2;;
	-c) input_file=$2; shift 2;;
	-b) BED=$2; shift 2;;
	-o) outprex=$2; shift 2;;
	--operation) operation=$2; shift 2;;
	--pseudocount) pseudocount=$2; shift 2;;
	--logx) logx=$2; shift 2;;
	--id) hasid=$2; shift 2;;
	--deeptools) multiBamSummary_operations=$2; shift 2;;
    -p) p=$2; shift 2;;
	-g) chromsize=$2; shift 2;;
	--tmp) tmpdir=$2; shift 2;;
	-v) verbose=T; shift 1;;
	-h) help; exit 1;;
    -*) break;;
     *) break;;
esac
done
#####################initiation
if [ -z "$in_file" ] || [ -z "$BED" ] || [ -z "$outprex" ]; then 
echo "error: 缺少参数 -b -t or -o. -h for help";exit 1;
fi

key=$(basename "$outprex")
mkdir -p $tmpdir 
if test -n "${outprex/$key}"; then mkdir -p ${outprex/$key}; fi

input_rpkm="$outprex.input.rpkm.bg"
treatment_rpkm="$outprex.rpkm.bg"
treatment_operated="$outprex.$operation.bg"

if test -r "/home/share/chrom_sizes/$chromsize.chrom.sizes"; then chromsize=/home/share/chrom_sizes/$chromsize.chrom.sizes
elif test -r "$chromsize"; then  chromsize=$chromsize
else echo "error: -g mission file. -h for help"; exit 1;
fi
awk -v OFS="\t" '{print $1,0,$2}' $chromsize > $tmpdir/$key.chrom.bed

#####################treatment rpkm
nohup multiBamSummary BED-file --BED "$BED" -b "$in_file" -o $tmpdir/$key.gz -p $p $multiBamSummary --outRawCounts $tmpdir/$key.tab #> $tmpdir/$key.log 

nohup multiBamSummary BED-file --BED $tmpdir/$key.chrom.bed -b "$in_file" -o $tmpdir/$key.chrom.gz -p $p $multiBamSummary --outRawCounts $tmpdir/$key.chrom.tab #> $tmpdir/$key.chrom.log 

awk -v OFS="\t" -v hasid=$hasid 'ARGIND==1{if(hasid>0){id[$1"\t"$2"\t"$3]=$hasid}} 
ARGIND==2&&FNR>1{total+=$4} 
ARGIND==3&&FNR==1{if(hasid>0){print "chr\tstart\tend\trpkm\tid"}else{print "chr\tstart\tend\trpkm"}}
ARGIND==3&&FNR>1{if(hasid>0){print $1,$2,$3,($4*1e9)/(($3-$2)*total,id[$1"\t"$2"\t"$3]}else{print $1,$2,$3,($4*1e9)/(($3-$2)*total}}' "$BED" $tmpdir/$key.chrom.tab $tmpdir/$key.tab > "$treatment_rpkm"

#####################input rpkm
if test -n "$input_file"; then
nohup multiBamSummary BED-file --BED "$BED" -b "$input_file" -o $tmpdir/$key.input.gz -p $p $multiBamSummary --outRawCounts $tmpdir/$key.input.tab #> $tmpdir/$key.input.log 

nohup multiBamSummary BED-file --BED $tmpdir/$key.chrom.bed -b "$input_file" -o $tmpdir/$key.input.chrom.gz -p $p $multiBamSummary --outRawCounts $tmpdir/$key.input.chrom.tab #> $tmpdir/$key.input.chrom.log 

awk -v OFS="\t" -v hasid=$hasid 'ARGIND==1{if(hasid>0){id[$1"\t"$2"\t"$3]=$hasid}} 
ARGIND==2&&FNR>1{total+=$4} 
ARGIND==3&&FNR==1{if(hasid>0){print "chr\tstart\tend\trpkm\tid"}else{print "chr\tstart\tend\trpkm"}}
ARGIND==3&&FNR>1{if(hasid>0){print $1,$2,$3,($4*1e9)/(($3-$2)*total,id[$1"\t"$2"\t"$3]}else{print $1,$2,$3,($4*1e9)/(($3-$2)*total}}' "$BED" $tmpdir/$key.input.chrom.tab $tmpdir/$key.input.tab > "$input_rpkm"
fi

#####################operation
if [ -r "$input_rpkm" ] && [ -r "$treatment_rpkm" ]; then
if [[ "$operation" = "log2ratio" ]]; then operation="logXratio"; logx=2
elif [ "$operation" = "log10ratio" ]; then operation="logXratio"; logx=10
elif [ "$operation" = "log2subtract" ]; then operation="logXsubtract"; logx=2
elif [ "$operation" = "log10rsubtract" ]; then operation="logXsubtract"; logx=10
fi

if [ "$operation" = "logXratio" ]; then
	awk -v OFS="\t" -v m=$pseudocount -v logx=$logx 'NR==1{print $0;next} NR==FNR{a[$1"\t"$2"\t"$3]=$4+m} NR>FNR{$4=log(($4+m)/a[$1"\t"$2"\t"$3])/log(logx);print $0}' "$input_rpkm" "$treatment_rpkm" > $treatment_operated
elif [ "$operation" = "ratio" ]; then
	awk -v OFS="\t" -v m=$pseudocount 'NR==1{print $0;next} NR==FNR{a[$1"\t"$2"\t"$3]=$4+m} NR>FNR{$4=($4+m)/a[$1"\t"$2"\t"$3];print $0}' "$input_rpkm" "$treatment_rpkm" > $treatment_operated
elif [ "$operation" = "reciprocal_ratio" ]; then
	awk -v OFS="\t" -v m=$pseudocount 'NR==1{print $0;next} NR==FNR{a[$1"\t"$2"\t"$3]=$4+m} NR>FNR{$4=($4+m)/a[$1"\t"$2"\t"$3];if($4<1)$4=-1/$4;print $0}' "$input_rpkm" "$treatment_rpkm" > $treatment_operated
elif [ "$operation" = "subtract" ]; then
	awk -v OFS="\t" 'NR==1{print $0;next} NR==FNR{a[$1"\t"$2"\t"$3]=$4} NR>FNR{$4=$4-a[$1"\t"$2"\t"$3];print $0}' "$input_rpkm" "$treatment_rpkm" > $treatment_operated
elif [ "$operation" = "logXsubtract" ]; then
	awk -v OFS="\t" -v m=$pseudocount -v logx=$logx 'NR==1{print $0;next} NR==FNR{a[$1"\t"$2"\t"$3]=$4} NR>FNR{if($4<a[$1"\t"$2"\t"$3]){$4=log(m)/log(logx)}else{$4=log($4-a[$1"\t"$2"\t"$3]+m)/log(logx)};print $0}' "$input_rpkm" "$treatment_rpkm" > $treatment_operated
fi
fi

#####################remove temp files
if [ -z "$verbose" ]; then rm $tmpdir/$key*;fi
